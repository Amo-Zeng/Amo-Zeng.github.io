import torch
from opacus import PrivacyEngine

model = Net()
optimizer = torch.optim.SGD(model.parameters(), lr=0.05)
privacy_engine = PrivacyEngine()

model, optimizer, data_loader = privacy_engine.attach(
    model,
    optimizer,
    data_loader,
    noise_multiplier=1.2,   # ε=1.2参数配置
    max_grad_norm=1.0,
